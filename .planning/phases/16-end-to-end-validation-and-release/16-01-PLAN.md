### Task 2: run-tests.sh extension

Add NEW Category 6 validator BEFORE the `# --- Main execution ---` section:

```bash
# Category 6: v2.0 category coverage (Phase 16 D-104)
validate_v2_category_coverage() {
    local file="$1"
    local name
    name="$(basename "$file")"

    # 13 v2.0 category anchor strings (one per category)
    declare -A v2_anchors=(
      [INTV]="decision_patterns"
      [BGRAPH]="business-graph.json"
      [DSGN]="agent-profiles.yaml"
      [ORCH]="(sequential|event-driven|conversational|loop|parallel)"
      [INTEG]="integration-manifest.yaml"
      [BROWSER]="(DISCOVERY-LICENSE-NOTICE|Patchright)"
      [DEPLOY]="DEPLOY-REPORT.md"
      [MEM]="(memory.md|state.json|last-run.json)"
      [RUNTIME]="correlation_id"
      [AUTON]="(autonomy|approval-router)"
      [MONITOR]="(jsonl-log-schema|briefing-agent)"
      [CTRL]="(activity-feed|status badges)"
      [ANTIC]="ANTICIPATED"
    )

    for category in INTV BGRAPH DSGN ORCH INTEG BROWSER DEPLOY MEM RUNTIME AUTON MONITOR CTRL ANTIC; do
        if grep -qE "${v2_anchors[$category]}" "$file"; then
            tap_ok "$name: v2.0 category $category covered"
        else
            tap_not_ok "$name: v2.0 category $category NOT covered" "Expected anchor: ${v2_anchors[$category]}"
        fi
    done
}
```

And insert the call in the main loop (after `validate_assertions`):

```bash
for scenario in "$SCENARIOS_DIR"/*.jsonl; do
    validate_json "$scenario"
    validate_fields "$scenario"
    validate_sequence "$scenario"
    validate_assertions "$scenario"
    validate_v2_category_coverage "$scenario"   # NEW
done
```

Target: run-tests.sh grows from 254 to ~290-310 lines.

**NOTE:** Bash 4+ associative arrays (`declare -A`) required. macOS ships Bash 3.2; Linux CI runs Bash 4+. The header `#!/usr/bin/env bash` already accepts whatever bash is on PATH; if macOS Bash 3.2 is the default, the validator will silently no-op on macOS. Workaround: switch to plain `case` statement matching anchor by category name (no associative array). This keeps macOS compatibility without losing functionality. Plan 16-01 Task 2 ships the case-statement variant for portability.
